char copiaAsciiPari(char *primaStringa, char *secondaStringa){  
    for (int i = 0; *primaStringa != '\0'; ++i) {  
        if (*primaStringa % 2 == 0){  
            *secondaStringa = *primaStringa;  
            secondaStringa++;  
        }  
        primaStringa++;  
    }  
    *secondaStringa = '\0';

}
